   122→### Flujo de Datos Completo
   123→
   124→```
   125→1. USUARIO INGRESA TAREA
   126→   ├─ Frontend: Captura objetivo
   127→   ├─ Validación: Verifica entrada
   128→   └─ API: Envía a backend
   129→
   130→2. BACKEND RECIBE TAREA
   131→   ├─ Autenticación: Verifica JWT
   132→   ├─ Validación: Valida datos
   133→   ├─ Almacenamiento: Guarda en DB
   134→   └─ Emit: Notifica inicio a frontend
   135→
   136→3. AGENTE PLANIFICADOR
   137→   ├─ Llama a OpenCode Go
   138→   ├─ Recibe plan de pasos
   139→   ├─ Descompone en subtareas
   140→   └─ Calcula dependencias
   141→
   142→4. AGENTE EJECUTOR
   143→   ├─ Ejecuta paso 1
   144→   ├─ Selecciona herramienta
   145→   ├─ Ejecuta herramienta
   146→   ├─ Almacena resultado
   147→   ├─ Emite log a frontend
   148→   └─ Repite para cada paso
   149→
   150→5. VERIFICADOR
   151→   ├─ Valida resultado
   152→   ├─ Si error: Reintentar
   153→   ├─ Si éxito: Siguiente paso
   154→   └─ Si crítico: Detener
   155→
   156→6. RESULTADO FINAL
   157→   ├─ Agrega todos los resultados
   158→   ├─ Formatea respuesta
   159→   ├─ Almacena en DB
   160→   └─ Emite a frontend
   161→
   162→7. FRONTEND MUESTRA RESULTADO
   163→   ├─ Detiene animación
   164→   ├─ Muestra resultado formateado
   165→   ├─ Ofrece opciones (descargar, repetir, etc.)
   166→   └─ Actualiza historial
   167→```

   415→#### 1. Agente Planificador
   416→
   417→```typescript
   418→class PlannerAgent {
   419→  private adapter: OpenCodeGoAdapter;
   420→  private memory: SemanticMemory;
   421→  
   422→  async createPlan(objective: string, constraints?: string[]): Promise<ExecutionPlan> {
   423→    // 1. Analizar objetivo
   424→    const analysis = await this.analyzeObjective(objective);
   425→    
   426→    // 2. Llamar a OpenCode Go para crear plan
   427→    const prompt = this.buildPlanningPrompt(objective, constraints, analysis);
   428→    const response = await this.adapter.chat([
   429→      { role: 'system', content: PLANNER_SYSTEM_PROMPT },
   430→      { role: 'user', content: prompt }
   431→    ]);
   432→    
   433→    // 3. Parsear plan
   434→    const plan = this.parsePlan(response);
   435→    
   436→    // 4. Validar plan
   437→    await this.validatePlan(plan);
   438→    
   439→    // 5. Guardar en memoria
   440→    await this.memory.store(objective, plan);
   441→    
   442→    return plan;
   443→  }
   444→  
   445→  private buildPlanningPrompt(objective: string, constraints?: string[], analysis?: any): string {
   446→    return `
   447→      Eres un planificador experto de tareas. Tu trabajo es descomponer objetivos complejos en pasos ejecutables.
   448→      
   449→      OBJETIVO: ${objective}
   450→      ${constraints ? `RESTRICCIONES: ${constraints.join(', ')}` : ''}
   451→      
   452→      Debes retornar un JSON con la siguiente estructura:
   453→      {
   454→        "steps": [
   455→          {
   456→            "number": 1,
   457→            "description": "Descripción del paso",
   458→            "tool": "nombre_herramienta",
   459→            "parameters": { ... },
   460→            "dependencies": [0],
   461→            "estimatedTime": 30
   462→          }
   463→        ],
   464→        "totalEstimatedTime": 120,
   465→        "riskFactors": ["factor1", "factor2"]
   466→      }
   467→    `;
   468→  }
   469→}
   470→```

   472→#### 2. Agente Ejecutor
   473→
   474→```typescript
   475→class ExecutorAgent {
   476→  private adapter: OpenCodeGoAdapter;
   477→  private tools: ToolRegistry;
   478→  private memory: WorkingMemory;
   479→  
   480→  async executePlan(plan: ExecutionPlan, execution: Execution): Promise<void> {
   481→    for (let i = 0; i < plan.steps.length; i++) {
   482→      const step = plan.steps[i];
   483→      
   484→      // 1. Preparar contexto
   485→      const context = await this.memory.getContext();
   486→      
   487→      // 2. Ejecutar herramienta
   488→      const tool = this.tools.get(step.tool);
   489→      const result = await tool.execute(step.parameters, context);
   490→      
   491→      // 3. Guardar resultado
   492→      execution.steps[i].result = result;
   493→      execution.steps[i].status = 'completed';
   494→      
   495→      // 4. Actualizar memoria
   496→      await this.memory.update(step.description, result);
   497→      
   498→      // 5. Emitir log
   499→      this.emitLog(`Paso ${i + 1} completado: ${step.description}`);
   500→    }
   501→  }
   502→}
   503→```

   505→#### 3. Agente Verificador
   506→
   507→```typescript
   508→class VerifierAgent {
   509→  private adapter: OpenCodeGoAdapter;
   510→  
   511→  async verifyStep(step: ExecutionStep, context: any): Promise<VerificationResult> {
   512→    // 1. Validar resultado
   513→    const isValid = this.validateResult(step.result, step.toolName);
   514→    
   515→    if (!isValid) {
   516→      // 2. Llamar a OpenCode Go para análisis
   517→      const analysis = await this.analyzeError(step, context);
   518→      
   519→      // 3. Determinar acción
   520→      if (analysis.canRetry) {
   521→        return { status: 'retry', reason: analysis.reason };
   522→      } else {
   523→        return { status: 'failed', reason: analysis.reason };
   524→      }
   525→    }
   526→    
   527→    return { status: 'success' };
   528→  }
   529→}

  1101→```typescript
  1102→// useWebSocket.ts
  1103→export function useWebSocket() {
  1104→  const socket = useRef<Socket>();
  1105→  const { setExecution, addLog } = useAppStore();
  1106→  
  1107→  useEffect(() => {
  1108→    socket.current = io(API_URL, {
  1109→      auth: {
  1110→        token: localStorage.getItem('token')
  1111→      }
  1112→    });
  1113→    
  1114→    // Escuchar eventos
  1115→    socket.current.on('execution:started', (execution) => {
  1116→      setExecution(execution);
  1117→    });
  1118→    
  1119→    socket.current.on('execution:step:started', (step) => {
  1120→      addLog({
  1121→        timestamp: new Date(),
  1122→        level: 'info',
  1123→        message: `Iniciando paso: ${step.description}`
  1124→      });
  1125→    });
  1126→    
  1127→    socket.current.on('execution:step:completed', (step) => {
  1128→      addLog({
  1129→        timestamp: new Date(),
  1130→        level: 'info',
  1131→        message: `Paso completado: ${step.description}`
  1132→      });
  1133→    });
  1134→    
  1135→    socket.current.on('execution:log', (log) => {
  1136→      addLog(log);
  1137→    });
  1138→    
  1139→    socket.current.on('execution:completed', (result) => {
  1140→      setExecution(result);
  1141→    });
  1142→    
  1143→    return () => {
  1144→      socket.current?.disconnect();
  1145→    };
  1146→  }, []);
  1147→  
  1148→  return socket.current;
  1149→}

  1586→Eres un planificador experto de tareas. Tu trabajo es:
  1587→1. Analizar el objetivo del usuario

  1589→3. Identificar dependencias entre pasos
  1590→4. Seleccionar herramientas apropiadas para cada paso
  1591→5. Estimar tiempo y recursos
  1592→
  1593→Retorna siempre un JSON con estructura:
  1594→{
  1595→  "steps": [
  1596→    {
  1597→      "number": 1,
  1598→      "description": "...",
  1599→      "tool": "...",
  1600→      "parameters": {...},
  1601→      "dependencies": [...],
  1602→      "estimatedTime": 30
  1603→    }
  1604→  ],
  1605→  "totalEstimatedTime": 120,
  1606→  "riskFactors": [...]
  1607→}
  1608→```
  1609→
  1610→### 2. Agente Ejecutor
  1611→
  1612→**Responsabilidad:** Ejecutar cada paso del plan
  1613→
  1614→**Proceso:**
  1615→1. Obtiene contexto actual
  1616→2. Ejecuta herramienta
  1617→3. Valida resultado
  1618→4. Actualiza memoria
  1619→5. Emite logs
  1620→6. Pasa al siguiente paso
  1621→
  1622→### 3. Agente Verificador
  1623→
  1624→**Responsabilidad:** Validar resultados y manejar errores
  1625→
  1626→**Proceso:**
  1627→1. Valida resultado de paso
  1628→2. Si error: Analiza y decide si reintentar
  1629→3. Si éxito: Continúa
  1630→4. Si crítico: Detiene ejecución
